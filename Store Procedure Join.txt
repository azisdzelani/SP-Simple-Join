ALTER PROCEDURE [dbo].[SP_View]
as

begin
	select p.ProductName, c.CategoryName, p.QuantityPerUnit, p.UnitPrice, c.Description  
	from Products p inner join Categories c
	on p.CategoryID = c.CategoryID
end
